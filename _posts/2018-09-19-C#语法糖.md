# 一些基础的C#语法糖

* 类的实例化的语法糖
~~~
Test test1 = new Test()
{
    Name = "45"
};
//等同于
Test test2 = new Test();
test2.Name = "45";
~~~

* 初始化List集合的值
~~~
List<Test> listTest1 = new List<Test>()
{
    new Test(){Name="1"},
    new Test(){Name="2"},
    new Test(){Name="3"},
};
//等同于
List<Test> listTest2 = new List<Test>();
listTest2.Add(new Test() { Name = "1" });
listTest2.Add(new Test() { Name = "1" });
listTest2.Add(new Test() { Name = "1" });
~~~

* 取list当中的值
~~~
listTest2.ForEach(test => Console.WriteLine(test.Name));
//等同于使用foeach写法
foreach (var item in listTest2)
{
    Console.WriteLine(item.Name);
}
~~~

* ?
~~~
string sex = "男";
int result = sex == "男" ? 1 : 0; //js,小程序等中用的比较多，当sex=="男"成立，则赋1值给result，不成立赋0给result
~~~

* ??
~~~
string name = null;
string isNull = name ?? "该值为空";
~~~

* 匿名类
~~~
var people = new
{
    Name = "123",
    Age  =  "12",
    //.....
};
~~~

* 方法参数默认值
~~~
//调用方法的时候，直接赋值也可以
Show();         //如果参数赋值了默认值，就可以不写参数
Show("56666");  //当然也可以赋值

//参数可以赋默认值
public static void Show(string name="56")
{
    Console.WriteLine(name);
}
~~~
